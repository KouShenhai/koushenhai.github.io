дҪ еҘҪе‘ҖпјҢжҲ‘зҡ„иҖҒжңӢеҸӢпјҒжҲ‘жҳҜиҖҒеҜҮпјҢж¬ўиҝҺжқҘеҲ°иҖҒеҜҮIoTдә‘е№іеҸ°пјҒ

# дёҖгҖҒSpring gRPC й…ҚзҪ®

Spring gRPCе®ҳж–№ең°еқҖ (opens new window)

# 1.1 Server з«Ҝй…ҚзҪ®пјҲspring.grpc.serverпјү

gRPC жңҚеҠЎжҸҗдҫӣиҖ…йңҖиҰҒй…ҚзҪ® Server з«ҜпјҢз”ЁдәҺжҡҙйңІ gRPC жҺҘеҸЈгҖӮ

# еұһжҖ§иҜҙжҳҺ

еұһжҖ§ зұ»еһӢ й»ҳи®ӨеҖј иҜҙжҳҺ
spring.grpc.server.port int 9090 gRPC жңҚеҠЎзӣ‘еҗ¬з«ҜеҸЈ
spring.grpc.server.servlet.enabled boolean true жҳҜеҗҰйҖҡиҝҮ Servlet е®№еҷЁжҸҗдҫӣ gRPC жңҚеҠЎгҖӮи®ҫдёә false ж—¶дҪҝз”ЁзӢ¬з«Ӣ Netty з«ҜеҸЈ

# й…ҚзҪ®зӨәдҫӢ

# laokou-distributed-id-segment жңҚеҠЎз«Ҝй…ҚзҪ®
spring:
  grpc:
    server:
      servlet:
        enabled: false   # зҰҒз”Ё Servlet жЁЎејҸпјҢдҪҝз”ЁзӢ¬з«Ӣ gRPC з«ҜеҸЈ
      port: 10111        # gRPC жңҚеҠЎзӣ‘еҗ¬з«ҜеҸЈ
  # йңҖиҰҒжіЁеҶҢgprc_portз«ҜеҸЈеҲ°NacosжңҚеҠЎеҸ‘зҺ°пјҢеҗҰеҲҷе®ўжҲ·з«Ҝж— жі•йҖҡиҝҮdiscovery://жңҚеҠЎеҗҚи®ҝй—®
  cloud:
	nacos:
	  discovery:
		metadata:
		  grpc_port: ${GRPC_PORT:10111}

иҜҙжҳҺпјҡservlet.enabled=false иЎЁзӨә gRPC иө°зӢ¬з«Ӣз«ҜеҸЈпјҲ10111пјүпјҢдёҺ HTTP ServletпјҲеҰӮ Tomcat зҡ„ 9094пјүдә’дёҚе№Іжү°пјҢз”ҹдә§зҺҜеўғжҺЁиҚҗжӯӨеҒҡжі•гҖӮ

# Maven дҫқиө–

<!-- gRPC Server з«Ҝ Starter -->
<dependency>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-grpc-server</artifactId>
</dependency>

# 1.2 Client з«Ҝй…ҚзҪ®пјҲspring.grpc.clientпјү

gRPC и°ғз”Ёж–№йңҖиҰҒй…ҚзҪ® Client з«ҜйҖҡйҒ“пјҲChannelпјүпјҢз”ЁдәҺиҝһжҺҘиҝңзЁӢ gRPC жңҚеҠЎгҖӮ

# еұһжҖ§иҜҙжҳҺ

еұһжҖ§ зұ»еһӢ й»ҳи®ӨеҖј иҜҙжҳҺ
spring.grpc.client.channels.<name>.address String вҖ” gRPC жңҚеҠЎең°еқҖ
spring.grpc.client.channels.<name>.enable-keep-alive boolean false жҳҜеҗҰејҖеҗҜ TCP Keep-Alive еҝғи·і
spring.grpc.client.channels.<name>.idle-timeout Duration вҖ” иҝһжҺҘз©әй—Іи¶…ж—¶ж—¶й—ҙ

ең°еқҖж јејҸиҜҙжҳҺпјҡ

  • жңҚеҠЎеҸ‘зҺ°пјҡdiscovery://жңҚеҠЎеҗҚпјҲй…ҚеҗҲ Nacos Loadbalancerпјү
  • зӣҙиҝһпјҡstatic://host:portпјҲејҖеҸ‘/жөӢиҜ•зҺҜеўғпјү

# й…ҚзҪ®зӨәдҫӢ

# laokou-auth / laokou-admin е®ўжҲ·з«Ҝй…ҚзҪ®
spring:
  grpc:
    client:
      channel:
        laokou-distributed-id-segment:                        # йҖҡйҒ“еҗҚз§°пјҲеҜ№еә”зӣ®ж ҮжңҚеҠЎеҗҚпјү
          target: discovery://laokou-distributed-id-segment   # йҖҡиҝҮ Nacos жңҚеҠЎеҸ‘зҺ°
		  keepalive:
			  time: 10s
			  timeout: 30s
		  idle:
			  timeout: 30s

# 2.2 Proto ж–Үд»¶зј–еҶҷ

# иҜӯжі•иҜҙжҳҺ

// 1. жҢҮе®ҡ proto3 иҜӯжі•пјҲеҝ…йЎ»ж”ҫеңЁз¬¬дёҖиЎҢпјү
syntax = "proto3";

// 2. Java д»Јз Ғз”ҹжҲҗйҖүйЎ№
option java_multiple_files = true;                          // жҜҸдёӘж¶ҲжҒҜз”ҹжҲҗзӢ¬з«Ӣ .java ж–Үд»¶
option java_package = "org.laokou.distributed.id.proto";    // Java еҢ…еҗҚ
option java_outer_classname = "DistributedIdProto";         // еӨ–еұӮеҢ…иЈ…зұ»еҗҚ

// 3. е®ҡд№үж¶ҲжҒҜдҪ“пјҲMessageпјү
message GenerateIdRequest {
    // з©әж¶ҲжҒҜдҪ“
}

message GenerateBatchIdRequest {
    int32 num = 1;          // еӯ—ж®өзј–еҸ·д»Һ 1 ејҖе§Ӣ
}

message GenerateIdResponse {
    string code = 1;        // зҠ¶жҖҒз Ғ
    string msg = 2;         // зҠ¶жҖҒж¶ҲжҒҜ
    int64 data = 3;         // з”ҹжҲҗзҡ„ ID
}

message GenerateBatchIdsResponse {
    string code = 1;
    string msg = 2;
    repeated int64 data = 3;  // repeated = Java List<Long>
}

// 4. е®ҡд№ү gRPC жңҚеҠЎжҺҘеҸЈ
service DistributedIdServiceI {
    rpc generateId(GenerateIdRequest) returns (GenerateIdResponse);
    rpc generateBatchIds(GenerateBatchIdRequest) returns (GenerateBatchIdsResponse);
}

# йҖүйЎ№иҜҙжҳҺ

йҖүйЎ№ еҖј иҜҙжҳҺ
java_multiple_files true жҜҸдёӘ message / service з”ҹжҲҗзӢ¬з«Ӣ .java ж–Үд»¶
java_package еҢ…еҗҚ з”ҹжҲҗзҡ„ Java д»Јз ҒеҢ…и·Ҝеҫ„
java_outer_classname зұ»еҗҚ еӨ–еұӮеҢ…иЈ…зұ»еҗҚпјҲеӯҳж”ҫж–Үд»¶жҸҸиҝ°з¬Ұзӯүе…¬е…ұе…ғж•°жҚ®пјү

# еёёз”Ёеӯ—ж®өзұ»еһӢ

Proto зұ»еһӢ Java зұ»еһӢ иҜҙжҳҺ
int32 int 32 дҪҚж•ҙж•°
int64 long 64 дҪҚж•ҙж•°
string String еӯ—з¬ҰдёІ
bool boolean еёғе°”еҖј
double double еҸҢзІҫеәҰжө®зӮ№
float float еҚ•зІҫеәҰжө®зӮ№
bytes ByteString еӯ—иҠӮж•°з»„
repeated T List<T> еҲ—иЎЁ/ж•°з»„

# 2.3 з”ҹжҲҗзҡ„ж–Үд»¶жё…еҚ•

GenerateIdResponse.java protoc е“Қеә”ж¶ҲжҒҜзұ»

GenerateBatchIdRequest.java protoc жү№йҮҸиҜ·жұӮж¶ҲжҒҜзұ»
